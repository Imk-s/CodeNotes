## 前缀和与差分

### 一维前缀和

给出数组n个数{1,3,2,5,7,3,8}，求l到r上所有数的和

用循环遍历时间复杂度较高，故把每个**前n项和**刻画成一个静态的数组，即前缀和

$sum[i]=sum[i-1]+a[i]$

$sum(l,r)=sum[r]-sum[l-1]$

```c++
for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        sum[i]=sum[i-1]+a[i];
    }
    int l,r;
    cin>>l>>r;
    cout<<sum[r]-sum[l-1]<<endl;
```

### 一维差分

给出数组n个数{1,3,2,5,7,3,8}，做m次询问，每次在区间[l,r]上所有数都加上k，求出变化后的数组

同样，为了提高效率，我们用查分数组diff[i]来表示第i个数与第i-1个数的**差值**，累加可得原数组，并且询问时只用动差分数组，非常高效

$diff[i]=a[i]-a[i-1]$

在l,r上加k，只需$diff[l]-=k;diff[r+1]-=k$

```c++
void add(int l,int r,int k){
    d[l]+=k;
    d[r+1]-=k;
}
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        add(i,i,a[i]); //相当于把查分数组初始化
    }
    cin>>m;
    for(int i=1;i<=m;i++){
        int l,r,k;
        cin>>l>>r>>k;
        add(l,r,k);
    }
    for(int i=1;i<=n;i++){
        d[i]+=d[i-1]; //把差分数组还原
        cout<<d[i]<<" ";
        }
    return 0;
}
```

### 二维前缀和

给一个二维数组，求出(x1,y1)到(x2,y2)这个矩形内所有数的和

重点在于公式，利用容斥原理：

预处理：$sum[i][j]=sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1]+a[i][j]$

查询：$sum=sum[x2][y2]-sum[x1-1][y2]-sum[x2][y1-1]+sum[x1-1][y1-1]$

```c++
int n,m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++) //下标从1开始，可以把第0行列置0
        {
            cin>>a[i][j];
            s[i][j]=s[i-1][j]+s[i][j-1]-s[i-1][j-1]+a[i][j];
        }
    
    int x1,x2,y1,y2;
    cin>>x1>>x2>>y1>>y2;
    cout<<s[x2][y2]-s[x1-1][y2]-s[x2][y1-1]+s[x1-1][y1-1]<<endl;
    return 0;
```

### 二维差分

对(x1,y1)到(x2,y2)矩形内的数加上k，需要一个差分数组diff来记录，公式为：

$d[x1][y1]+=k;$   $d[x2+1][y1]-=k;$   $d[x1][y2+1]-=k;$   $d[x2+1][y2+1]+=k;$

然后，通过求前缀和的方式求(x,y)的值：

$d[i][j]=d[i-1][j]+d[i][j-1]-d[i-1][j-1]+d[i][j]$

```c++
void add(int x1,int y1,int x2,int y2,int k){
    d[x1][y1]+=k;
    d[x2+1][y1]-=k;
    d[x1][y2+1]-=k;
    d[x2+1][y2+1]+=k;
}
int main(){
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        {
            cin>>a[i][j];
            add(i,j,i,j,a[i][j]);
        }
    int x1,y1,x2,y2,k;
    cin>>x1>>y1>>x2>>y2>>k;
    add(x1,y1,x2,y2,k);
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++)
        {
            d[i][j]+=d[i-1][j]+d[i][j-1]-d[i-1][j-1];
            cout<<d[i][j]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
```

```c++
vector<int> add(vector<int>&a,vector<int>&b){
    if(a.size()<b.size())swap(a,b);//保证b是小位
    vector<int>c;
    int t=0;//t代表当前的进位
    for(int i=0;i<a.size();i++){

        if(i<b.size())t+=b[i];
```

### 高精度减法

1.保证A>=B,否则输出符号

2.不够减则向高位借1

3.减法后若出现前导0则需去除

```c++
bool cmp(vector<int>&a,vector<int>&b){
    if(a.size()!=b.size())return a.size()>b.size();
    for(int i=a.size()-1;i>=0;i--)
        if(a[i]!=b[i])return a[i]>b[i];
    return 1;
}
vector<int> sub(vector<int>&a,vector<int>&b){
    if(!cmp(a,b))swap(a,b);
    vector<int>c;
    int t=0; //借位
    for(int i=0;i<a.size();i++){
        t=a[i]-t;
        if(i<b.size())t-=b[i];
        c.push_back((t+10)%10); //防止t为负
        if(t<0)t=1; //借位
        else t=0;
    }
    while(c.size()>1&&c.back()==0)c.pop_back();
    return c;
}
```

### 高精度乘法

高精度×低精度(大整数a*int b)

把b看作一个整体，参与a的每一位运算，并且处理进位

```c++
vector<int> mul(vector<int>&a,int b){
    vector<int>c;
    int t=0; //进位
    for(int i=0;i<a.size()||t;i++){ //循环条件包括t!=0
        if(i<a.size())t+=a[i]*b;
        c.push_back(t%10);
        t/=10;
    }
    while(c.size()>0&&c.back()==0)c.pop_back();
    return c;
}
```

### 高精度除法

高精度 / 低精度 ，由于除法是从最高位开始算的，所以最后需要反转

```c++
vector<int> div(vector<int>&a,int b){
    vector<int>c;
    int r=0; //t为余数
    for(int i=a.size()-1;i>=0;i--){
        r=r*10+a[i];
        c.push_back(r/b);
        r%=b; //更新余数
    }
    reverse(c.begin(),c.end()); //反转
    while(c.size()>1&&c.back()==0)c.pop_back();
    return c;
}
```



## 二分查找

二分查找某个数在数组的位置，首先进行sort排序，然后利用递归。

```c++
int bsearch(int l,int r,int x){
    if(l>r)return -1;
    int mid=(l+r+1)/2;
    if(a[mid]==x)return mid+1; //数组从0开始
    else if(a[mid]>x) return bsearch(l,mid-1,x);
    else return bsearch(mid+1,r,x);
    //当然x可以放到全局变量
}
```

最常用的是内置函数：

`upper_bound(a,a+n,val)`:第一个>val的元素的位置，返回类型是**迭代器**(指针)

`lower_bound(a,a+n,val)`:第一个>=val的元素的位置

常见用法：

```c++
int pos=upper_bound(a+1,a+n+1,val)-a; //如果下标从1开始，要+1,减去首元素地址a用来转化成位置
auto it=upper_bound(v.begin(),v.end(),val);
int pos=it-v.begin();
```

若是找第一个<x的元素：lower_bound()-1

若是找第一个<=x的元素：upper_bound()-1

## *map*排序

std::map和std::unordered_map是按照Key值自动升序排序的，如果想对其特殊排序，有以下的处理技巧：

### map按结构体排序

map无法直接用sort，所以当开map<string,student>的时候，若想对结构体排序，需要将map的结构体放入vector数组，再对数组sort排序。

下面以具体的例子讲解：

一场比赛中，给出一个滚榜，每个学生有对应的学号，还有对应的得分记录，一个学生可能有多次得分，求最后的排名，如果每次都用结构体数组查找，那么时间复杂度会变得很大，所以用哈希表。

如果最后要求输入id查找排名，就再开一个哈希映射，把排名映射回去

```c++
struct student{
	string id;
	int sc=0;
};
bool cmp(student s1,student s2){
	return s1.sc!=s2.sc?s1.sc>s2.sc:s1.id<s2.id;
}
unordered_map<string,student> mp;
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		string s;
		int score;
		cin>>s>>score;
		mp[s].id=s;
		mp[s].sc+=score; //进行映射查找 
	}
	vector<student> ans;
	for(auto &it:mp){
		ans.push_back(it.second); //倒入ans数组 用second，因为取出的it实际上是一个pair 
	}
	sort(ans.begin(),ans.end(),cmp);
	for(auto &it:ans){
		cout<<it.id<<" "<<it.sc<<endl;
	}
	unordered_map<string,int> rankmp;
	for(int i=0;i<ans.size();i++) rankmp[ans[i].id]=i+1;
	string name;
	cin>>name;
	cout<<rankmp[name]<<endl;
	return 0;
}
```

### map按value排序

map默认按key排序，如果想按照value排序，应该利用**存放pair的vector数组**，将map转化为pair，再进行排序。

同样，我们以具体的例子理解：

给出一个商店商品的出现频次，按商品销售量从高到低，同销售量名称字典序从小到大的顺序排序

```c++
unordered_map<string,int> mp;
bool cmp(pair<string,int>&p1,pair<string,int>&p2){
	return p1.second==p2.second?p1.second>p2.second:p1.first<p2.first;
}
int main(){
	int n;
	cin>>n;
	while(n--){
		string name;
		cin>>name;
		mp[name]++; //统计出现次数 
	}
	vector<pair<string,int>>vec(mp.begin(),mp.end()); //把哈希表拷贝到数组中
	sort(vec.begin(),vec.end());
	for(auto&it:vec){
		cout<<it.first<<" "<<it.second<<endl; 
	}
	return 0;
}
```

### 独立性检验

要查找一句话的的某个单词，不能光用find找，因为这样有可能把以这个单词为子串的大单词也包括进去，所以应该利用isalnum函数(判断是否是数字或字母),处理时，先把字符串左右各加一个空格“ ”,这样可以省去很多边缘条件的判断.

```c++
string s;
bool check(int pos,int len){
	if(!isalnum(s[pos-1])&&!isalnum(s[pos+len]))return 1;
	return 0;
}
int main(){
	getline(cin,s);
	s=" "+s+" ";
	string target;
	cin>>target;
	int pos=0,cnt=0;
	int len=target.size();
	while((pos=s.find(target,pos))!=string::npos){ //pos的赋值和查找合为一体 
		if(check(pos,len))cnt++;
		pos+=len;
	}
	cout<<cnt<<endl;
	return 0;
}
//后面要操作时，一定要注意下标问题！！空格会导致偏移！！
```

# 贪心

## *LIS*最长上升子序列(*Dilworth*定理)

一个数组中按下标次序选取子序列，递增/递减的子序列的长度

Dilworth定理就是说：把整个序列用单调序列覆盖所需要的链数是其反链的个数，比如至少取几个递增子序列才能把整个数组取遍，个数就是这个序列**最长不下降子序列**的长度.

在贪心二分解法中，用d[]数组表示长度问len的上升子序列中末尾元素的最小值，流程是：

1.如果a[i]大于d[]数组的最后一个元素，说明可以接在最长子序列的后面，长度len+1

2.如果不是，就找到d数组前面第一个>=a[i]的数，并替换

通过这样可以将时间复杂度降低到O(nlogn)

证明利用反证法，不详细展开

```c++
for(int i=1;i<=n;i++){
    if(len==0||a[i]>d[len])d[++len]=a[i];
else{
    int pos=lower_bound(d+1,d+len+1,a[i])-d;
    d[pos]=a[i];
    }
}
cout<<len<<endl;
```

## 素数筛法

### 埃氏筛法

开一个布尔数组表示是否是合数(被筛掉的)，从i*i开始筛去

```c++
typedef long long ll;
bool st[1000];
int primes[1000],cnt; //primes存素数，cnt可以存个数
void getprime(int n){
    for(int i=2;i<=n;i++){
        if(!st[i]){
            primes[cnt++]=i; //如果没被筛掉，存入素数数组
            for(ll j=(ll)i*i;j<=n;j+=i){
                st[j]=1; //筛掉
            }
        }
    }
}
```



### 欧拉筛法

同上，只不过为了保证每个合数只被他的最小质因数筛掉，需要做一些改动

```c++
typedef long long ll;
bool st[1000];
int primes[1000],cnt; //primes存素数，cnt可以存个数
void getprime(int n){
    for(int i=2;i<=n;i++){
        if(!st[i])primes[cnt++]=i;
        for(int j=0;primes[j]<=n/i;j++){
            st[primes[j]*i]=1;
            //如果i除primes[j]为0，说明i*priems[j]最小因子就是primes[j]
            //继续筛会导致priems[j+1]被筛掉，所以break
            if(i%primes[j]==0)break;
        }       
    }
}
```

## 快速幂

给出一个a，求a的k次方，如何快速地求解？答案取p的模

通过位运算，每次右移一位，如果最低二进制是1，把倍增结果乘进去.

```c++
typedef long long ll;
ll qsm(ll a,ll k,ll p){
    ll res=1; //结果
    a%=p; //防止a一开始就比p大
    while(k){
        if(k&1)res=res*a%p; //如果只是求结果，不用%p
        a=a*a%p;
        k>>=1; //k右移一位处理下一位二进制
    }
    return res;
}
```



# 数据结构

## 单调栈

寻找每个数左/右边第一个比它大/小的数：使用手写栈的方式，使栈中元素保存单调

例如，找一个数组中每个数右边第一个比它大的数的下标，就保持栈单调递减

```c++
const int N=3e6+10;
int ans[N],a[N]; //a是原数组,ans存答案
int stk[N],t; //stk是栈，t是指向栈顶元素的指针
int main(){  
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)cin>>a[i];
    t=0; //清空栈
    for(int i=1;i<=n;i++){
        while(t>0&&a[stk[t]]<a[i]){ //a[i]比栈顶大就操作，维持递减
            ans[stk[t]]=i;
            t--; //出栈
        }
        stk[++t]=i; //当前元素入栈
    }
    while(t>0){ //如果最后没有比栈顶还大的元素，栈不为空，说明找不到
        ans[stk[t]]=0;
        t--;
    }
    for(int i=1;i<=n;i++)cout<<ans[i]<<" ";
    return 0;
}
```

## 单调队列

给一个数组，给一个长度为l的窗口，求这个窗口在滑动过程中的最大/最小值

以滑动窗口求最大值为例，操作如下：

1.入队前去除尾部：如果队尾元素比当前元素小，那么它就不可能成为最大，把它踢出去

2.入队

3.检查队头，如果下标超过窗口，也要去除

4.记录答案，即是队头元素

```c++
const int N=3e6+10;
int a[N]; //a是原数组,ans存答案
int q[N],tt,hh; //q是队列,存下标，tt是队尾，hh是队头
int main(){  
    int n,k;
    cin>>n>>k;
    for(int i=1;i<=n;i++)cin>>a[i];
    hh=0,tt=-1; //初始化
    for(int i=1;i<=n;i++){
        if(hh<=tt&&q[hh]<i-k+1)hh++; //在[i-k+1,i]上枚举，如果队头下标比i-k+1还小，就过期了
        while(hh<=tt&&a[q[tt]]<=a[i])tt--;
        q[++tt]=i;
        if(i>=k)cout<<a[q[hh]]<<" ";
    }
    return 0;
}
```

## *DFS*深度优先搜索

*DFS*，深度优先搜索，通俗来讲就是**不撞南墙不回头**，如果一个点的周边有几个点与其向“连通”，先选择其中一个点，再找到这个点的连通点，这样不断地**顺藤摸瓜**，直到找到最后只有自己的终点，就走完了一条**路**，然后回溯到上一个点，就这样不断重复递归，最后可以把**所有的路**都走一遍

dfs通用流程模板如下：

```c++
void dfs(当前状态){
	if(到达终点/边界){
	记录答案；
	return;
	}
	for(枚举当前状态可以产生的下一个分支){
	if(分支合法){
	标记改分支被占用;
	dfs(下一步状态)
	取消标记;
	}
	}
}
```

下面给出最典型的几个例子：

### 遍历迷宫所有路

给出一个n×m的迷宫，给出起点和终点，以及中间某些格点不能走，每次只能走上下左右，问一共有多少条路径

```c++
const int N=50;
int n,m;
int sx,sy,ex,ey; //起点和终点
int g[N][N];
bool vis[N][N]; //标记是否走过
int ans=0;
int dx[5]={0,-1,1,0,0};
int dy[5]={0,0,0,-1,1};
bool check(int x,int y){
    if(x<=0||x>n||y<=0||y>m)return 0; //检查是否越界
    else if(vis[x][y]||g[x][y]!=0) return 0; //检查是否已经走过，是否有墙壁
    return 1;
}
void dfs(int x,int y){
    if(x==sx&&y==sy)vis[x][y]=1; //注意，如果传进来的是起点，一定要把起点先标记了！！
    if(x==ex&&y==ey){
        ans++;
        return;
    }
    for(int i=1;i<=4;i++){
        int tx=x+dx[i];
        int ty=y+dy[i];
        if(check(tx,ty)){
            vis[tx][ty]=1; //先标记
            dfs(tx,ty);
            vis[tx][ty]=0; //再回溯
        }
    }
}
int main(){ 
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            cin>>g[i][j];
    cin>>sx>>sy>>ex>>ey;
    dfs(sx,sy);
    cout<<ans<<endl;
    return 0;
}
```

## *BFS*广度优先搜索

*BFS*,深度优先搜索，就是优先把一个点连接的所有点都访问一遍，以此类推，形成**水的扩散**般的效果，求出最短路等等,需要借助**队列**实现,需注意BFS中一定不用**回溯**.

BFS通用流程模板如下：

```c++
void bfs(){
	queue<int> q;
	将起点压入队列
	标记起点的vis[]为true，记录step[起点]=0
	while(队列非空){
	取出队头元素u=q.front();
	弹出队头元素q.pop();
	if(u是终点)return step[u];
	for(枚举u能走到的所有下一步v){
		if(v合法，没越界，不是墙壁，没有被访问过){
		vis[v]=true;
		step[v]=step[u]+1;
		将v压入队列q，等待下一次处理
		}
	}
	}
	如果队列空了还没有找到，说明无法到达终点
}
```

### 迷宫最短路

跟DFS情境一样，只不过这次是求起点到终点的最短路需要几步

```c++
const int N=150;
int n,m;
int g[N][N];
bool vis[N][N];
int dis[N][N];
int dx[5]={0,-1,1,0,0};
int dy[5]={0,0,0,-1,1};
struct node{
    int x,y; //结构体存坐标
};
bool check(int x,int y){
    if(x<=0||x>n||y<=0||y>m)return 0;
    else if(g[x][y]!=0||vis[x][y])return 0;
    return 1;
}
int bfs(int sx,int sy,int ex,int ey){
    queue<node> q;
    q.push({sx,sy});
    vis[sx][sy]=1;
    dis[sx][sy]=0; //起点到起点步数为0
    while(!q.empty()){
        node cur=q.front();
        q.pop();
        if(cur.x==ex&&cur.y==ey)return dis[cur.x][cur.y];
        for(int i=1;i<=4;i++){
            int tx=cur.x+dx[i];
            int ty=cur.y+dy[i];
            if(check(tx,ty)){
                vis[tx][ty]=1;
                dis[tx][ty]=dis[cur.x][cur.y]+1;
                q.push({tx,ty});
            }
        }
    }
    return -1; //如果没找到输出-1
}
```

### 一维数轴BFS

以**洛谷P1588 Catch The Cow S**为例，给出数轴上起点x和终点y，每次操作将x+=1或-=1或*=2，求x到y最短几次变化

```c++
const int N=1e5+5;

    q.push(x);
    vis[x]=1;
    dis[x]=0;

        int u=q.front();
        q.pop();
        if(u==y)return dis[u];

                dis[next[i]]=dis[u]+1;
                q.push(next[i]);
            }
        }
    }
    return -1; //如果没找到输出-1
}
int main(){
    cin>>n;

    memset(dis,0,sizeof(dis));

}
    return 0;
}
```

## 背包问题

### 01背包

有N个物品和一个容量为V的背包，第i件物品的体积是w[i]，价值是v[i]，每件物品只能放一次，将哪些物品放入背包可使总价值最高

用dp[j]表示当前容量为j时所能获得的最大价值

方程：$dp[j]=max(dp[j],dp[j-w[i]]+v[i])$

1.不选第i件，则还是dp[i]

2.若选第i件，那么就是dp[j-w[i]]+v[i]

3.体积必须从大到小逆序遍历更新，如果正序，前面更新过的dp值会重复利用，导致一件物品被选了多次

```c++
for(int i=1;i<=n;i++)
    for(int j=V;j>=w[i];j--)
        dp[j]=max(dp[j],dp[j-w[i]]+v[i]);
cout<<dp[V]<<endl;
```

## 线性DP

### *LIS*最长上升子序列

上文已经给出lis的贪心解法，这里的dp解法时间复杂度是O($n^2$)

dp[i]表示以第i个数字a[i]结尾的最长子序列的长度

对于每个i，枚举它前面所有的j，如果a[j]<a[i]，说明a[i]可以接在它后面，所以方程是：

$dp[i]=max(dp[i],dp[j]+1)$

```c++
for(int i=1;i<=n;i++){
    dp[i]=1; //一定要记得初始化
    for(int j=1;j<i;j++){
        if(a[j]<a[i]) dp[i]=max(dp[i],dp[j]+1);
    }
    res=max(res,dp[i]);
}
cout<<res<<endl;
```

###  *LCS*最长公共子序列

求两个序列中共有的，保持原顺序的最长的子序列，时间复杂度O(N*M)

用dp[i] [j]表示序列A的前i个元素和B的前j个元素的最长lcs的长度

1.如果a[i]==b[j]说明当前字符可以加入序列，$dp[i][j]=dp[i-1][j-1]+1$

2.如果不相等，则二者退一步取最大(最大值是可以有重叠的)

$dp[i][j]=max(dp[i-1][j]+dp[i][j-1])$

```c++
for(int i=1;i<=n;i++){
    for(int j=1;j<=m;j++){
        if(a[i]==b[j]) dp[i][j]=dp[i-1][j-1]+1;
        else dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
    }
}
cout<<dp[n][m]<<endl;
```

### *LCIS*最长上升公共子序列

求两个序列中共有且递增的子序列最大长度

dp[j]表示以序列B的**第j个元素b[j]结尾**的lcis的最大长度

外层循环枚举A的i个元素

内层循环维护：在B遍历的同时维护一个遍历val，表示b[1~j-1]中严格小于a[i]的lcis最大长度

1.当a[i]==b[j]时，找到了一个公共元素可以接在val后面：$dp[j]=val+1$

2.当a[i]>b[j]时，说明b[j]有资格作为为了接在a[i]上，更新：$val=max(val,dp[j])$

```c++
for(int i=1;i<=n;i++){
    int val=0;
    for(int j=1;j<=n;j++){
        if(a[i]==b[j])dp[j]=max(dp[j],val+1);
        if(a[i]>b[j])val=max(val,dp[j]);
    }
}
int res=0;
for(int i=1;i<=n;i++)
    res=max(res,dp[i]);
cout<<res<<endl;
```

### DFS序/BFS序遍历

DFS序是最常见的，以两个P2169为例

## 二叉树由前/后+中序遍历建树

给出二叉树的前/后序遍历和中序遍历，可以唯一确定这可棵二叉树，如何用代码实现：**递归**

中序序列可以用一个哈希表存每个元素出现的下标，然后通过前/后序序列找到第一个根节点，就可以直接找根节点对应的中序序列的下标，再将其左右递归。

先找到根节点(原始祖宗)的下标k

左右子树的建树是关键，中序遍历传入的比较简单，都是**(il,k-1)(左子树),(k+1,ir)(右子树)**

后续遍历传入的需要重点记忆：

leflen=k-il代表了左子树的长度

前序：(**pl+1**,pl+leflen)(左子树),(pl+leflen+1,pr)(右子树)

后序：(pl,pl+leflen)(左子树),(pl+leflen+1,**pr-1**)(右子树)

可以发现：中间分界点都是pl+leflen,pl+leflen+1，但是由于前序遍历pl是根节点，后序遍历pr是根节点，为了防止无限递归，需要**把根节点排除在外**.

### 前序+中序

前序序列第一个肯定是根节点，拿这个根节点对应到中序序列中，再进行左右建树，递归：

```c++
int pre[110],post[110],in[110];
unordered_map<int,int> pos; //存中序遍历每个数的下标 
struct nod{
	int val;
	nod* lef=NULL;
	nod* rig=NULL;
};
nod* build(int il,int ir,int pl,int pr){
	if(pl>pr)return nullptr; //区间为空，说明没有节点了
	int root=post[pr]; //前序遍历pl为根
	nod *p=new nod;
	p->val=root;
	int k=pos[root]; //找到下标
	int leflen=k-il; //左子树的长度
	p->lef=build(il,k-1,pl+1,pl+leflen);
	p->rig=build(k+1,ir,pl+leflen+1,pr); //递归构建左右子树，一定要注意pl+1!,要把根节点排除在外，否则无限递归！ 
	return p; 
}
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)cin>>post[i];
	for(int i=1;i<=n;i++){	
	cin>>in[i];
	pos[in[i]]=i;
	}
	nod * p=build(1,n,1,n);
	return 0;
}
```

### 链式前向星

## 拓扑排序

在图中每一个节点的前置节点都在它前面的排序就叫做拓扑排序.

拓扑排序必要条件是**有向无环图(DAG)**,为什么有向？因为节点前置条件依赖于图的箭头方向，为什么无环？因为有环会导致循环依赖.

算法核心流程是：

遍历找入度为0的点作为起点，然后进行BFS，并把入度减一(消除影响)

```c++
vector<int> g[MAXN];
int indegree[MAXN]; //存放每个节点的入度
vector<int> topo;
bool toposort(int n){ //传入节点总数
    queue<int>q;
    for(int i=1;i<=n;i++){
        if(indegree[i]==0)q.push(i);
    }
    while(!q.empty()){
        int x=q.front();
        q.pop();
        topo.push_back(x); //拓扑入列的正确时机是一个节点被出队时，否则如果一开始节点入度为零，永远不会入队
        for(auto i:g[x]){
            indegree[i]--;
            if(!indegree[i]){
                q.push(i);
            }
        }
    }
    return topo.size()==n; //这个是判断是否能排序，如果成环或非有向，最后节点数回比n小
}
```

## 最小生成树

在**无向带权图**中选择一些边，在保证**连通性**的情况下，使得边的**总权值**最小

### *kruskal*算法

把所有的边从小到大排序，如果会形成环就不选择当前的边，如果不会，就选择，这样只需要一个并查集就行了.

```c++
const int N=5010; //最大点数
const int M=2e5+10; //最大边数
int fa[N];
int n,m;
struct edge{
    int v;
    int u;
    int w;
    bool operator<(const edge&x)const{
        return w<x.w;
    }
}ed[M];
void init(int n){
    for(int i=1;i<=n;i++)fa[i]=i;
    return;
}
int find(int x){
    if(fa[x]!=x)fa[x]=find(fa[x]);
    return fa[x];
}
bool merge(int x,int y){
    int rx=find(x);
    int ry=find(y);
    if(rx!=ry){
        fa[rx]=ry;
        return 1;
    }
    else return 0;
}
int kruskal(){
    init(n);
    sort(ed+1,ed+1+m);
    int ans=0,cnt=0;
    for(int i=1;i<=n;i++){
        if(merge(ed[i].u,ed[i].v)){
            ans+=ed[i].w;
            cnt++;
        }
        if(cnt==n-1)break;
    }
    if(cnt==n-1)return ans;
    else return -1; //如果最后没有等于n-1，说明无法找出最小生成树
}
```

### *Prim*算法

1.解锁的点的集合叫set，解锁的边叫heap(小跟堆)，初始都为空

2.从任意点开始加入到set，把它所有的边加入到小跟堆里去

3.弹出heap中最小的边，看看它去往的点是什么，如果已经在set里，重复再进行一次，如果不在，说明这条边在最小生成树里，把点加入set

这里代码跟dijkstra基本上没有区别

```c++
const int M=4e5+10;
const int INF=0x3f3f3f;
struct edge{
    int to;
    int wei;
};
struct node{
    int id;
    int dis;
    bool operator<(const node&x)const{
        return dis>x.dis;
    }
};
int dis[M];
bool vis[M];
int n,m;
vector<edge> adj[M];
int prim(int start,int n){
    for(int i=1;i<=n;i++)dis[i]=INF;
    dis[start]=0;
    int ans=0,cnt=0;
    priority_queue<node>q;
    q.push({start,0});
    while(!q.empty()){
        node cur=q.top();
        q.pop();
        int u=cur.id;
        if(vis[u])continue;
        vis[u]=1;
        cnt++;
        ans+=cur.dis;
        for(auto i:adj[u]){
            int v=i.to;
            int w=i.wei;
            if(!vis[v]&&w<dis[v]){
                dis[v]=w;
                q.push({v,dis[v]});
            }
        }
    }
     if(cnt==n)return ans;
     else return -1;
}
```

## 最短路问题

### *Dijkstra*算法(单源最短路)

求一个**有向图**中一点到任意一点的最短路径，边的权值没有**负数**

这里直接用堆结构实现：

```c++
#define int long long //防止溢出，根据题目选择
const int N=1e5+5;
const int INF=2147483647;
struct edge{
    int to;
    int wei;
};
struct node{
    int id;
    int dist;
    bool operator<(const node&x)const{
        return dist>x.dist;
    }
};
int dis[N];
int vis[N];
vector<edge> adj[N];
void dijkstra(int start,int n){
    priority_queue<node>q;
    for(int i=1;i<=n;i++){
        dis[i]=INF;
        vis[i]=0;
    }
    dis[start]=0;
    q.push({start,0});
    while(!q.empty()){
        node cur=q.top();
        q.pop();
        int u=cur.id;
        if(vis[u])continue;
        vis[u]=1;
        for(auto i:adj[u]){
            int v=i.to;
            int w=i.wei;
            if(dis[u]+w<dis[v]){
                dis[v]=dis[u]+w;
                q.push({v,dis[v]});
            }
        }
    }
    return;
}
```

### *Floyd*算法

求图中任意两点的最短距离，时间复杂度O(n^3 )，空间复杂度O(n^2),可以适用于任何图，不管有向还是无向，不管边权是正还是负，但是不能有**负环**(负环就是当一个环上有负数且负数比正数大的时候，可以通过无限次循环这个环，使总距离一直减小，最短路就不存在了)

外层先枚举每个中间点桥梁，然后内两层枚举点，这样可以保证每个节点都是"接力棒"

```c++
typedef long long ll;

                if(i==j)dis[i][j]=0;
                else dis[i][j]=INF; //初始化
            }
}
void floyd(){
    for(int k=1;k<=n;k++)
        for(int i=1;i<=n;i++)
            for(int j=1;j<=n;j++)
                {

                }
}
```

## st表

st表是解决**RMQ问题(区间最值)**的利器，利用了倍增的思想，且查询复杂度为O(1)，但是**不支持修改**

方程：$st[i][j]=max(st[i][j-1],st[i+1<<(j-1)][j-1])$

$st[i][j]$ 表示从下标i开始，区间长度为$2^j$的区间内的最大值

```c++
const int N=100010;
const int M=20;
int a[N];
int st[N][M];
void init(int n){
    for(int i=1;i<=n;i++)st[i][0]=a[i];
    for(int j=1;j<M;j++)//外存循环枚举区间长度
        for(int i=1;i+(1<<j)-1<=n;i++)//内层循环枚举区间的起点
            st[i][j]=max(st[i][j-1],st[i+(1<<(j-1))][j-1]);
}
int query(int l,int r){
    int k=__lg(r-l+1);//gcc内置函数，表示log以2为底向下取整
    return max(st[l][k],st[r-(1<<k)+1][k]);
}
```

## 逆序对

给一个数组a，逆序对就是其中i<j但是a[i]>a[j]的一对数，逆序对的总对数就是**只交换相邻两个数**，求出至少需要换几次才能将**序列变得有序**.

1.离散化，将序列变成下标

2.利用树状数组，维护一个**频率桶数组**，add(id,i)表示编号为id的数出现次数+1,query(x)就是编号<=id的数总共出现了几次.

```c++
int n,a[N],tr[N];
vector<int> nums;
int getid(int x){
    return lower_bound(nums.begin(),nums.end(),x)-nums.begin()+1;
}
int lowbit(int x){
    return x&-x;
}
void add(int x,int v){
    for(int i=x;i<=n;i+=lowbit(i))
        tr[i]+=v;
}
int query(int x){
    int res=0;
    for(int i=x;i;i-=lowbit(i))
        res+=tr[i];
    return res;
}
int main(){ 
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        nums.push_back(a[i]);
    }
    sort(nums.begin(),nums.end());
    ll ans=0;
    nums.erase(unique(nums.begin(),nums.end()),nums.end());
    for(int i=1;i<=n;i++){
        int id=getid(a[i]);
        int x=query(id);
        ans+=(i-1)-x;
        add(id,1);
    }
    cout<<ans<<endl;
    return 0;
}
```
